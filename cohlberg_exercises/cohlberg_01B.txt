# cohlberg_01B.txt, WESmith 04/01/23

# if showing hydrogens is desired, it must be set before loading the object
#set pdbAddHydrogens;

load =1PGB;

delete water;

show info;
show sequence;
show structure;

cartoon only;
color structure;

pause 'cartoon view, hit "resume" to continue'

backbone only;
backbone 100;
#color group;

# stick figure of hbonds
hbonds calculate;
hbonds 40;
color hbonds orange;
set hbonds backbone;

pause 'this is a stick figure of the H bonds, type "resume" to see a more detailed picture"

# more detailed atomic structure of hbonds
select all;
wireframe 50;
spacefill 100;
color amino;
set hbonds sidechain;
#color hbonds cpk;
#color hbonds orange;

pause 'H bonds visible, residues color-coded, type "resume" to isolate the beta sheet'

# isolate the sheet
select sheet;
color cpk;
restrict sheet;
cartoon 200;
#color yellow;

# turn off hbonds in helix
select not sheet;
hbonds off;

pause 'type "resume" to isolate the alpha helix'

#isolate the helix
select all;
wireframe 50;
spacefill 100;
color cpk;
restrict helix;
cartoon 200;
select helix;
hbonds 40;
select not helix;
hbonds off;

pause 'type "resume" to see all of the H bonds in isolation"

select all;
hbonds 40;
color hbonds orange;
wireframe off;
spacefill off;
cartoon off;

pause 'type "resume" to see the cartoon with the H bonds"

cartoon;
color structure;
